  15 template<typename T> using CConstIterBase = std::iterator<std::bidirectional_iterator_tag, const T, std::ptrdiff_t, const T *, const T &>;
  16 
  17 template<typename T>
  18 struct nsSdD::CList<T>::CConstIterator : public CConstIterBase<T>
  19 {
  20 private:
  21     /*!
  22        @var node
  23        @brief The current node of the iterator.
  24    */
  25     CNodePtr node;
  26 
  27 public:
  28     /*!
  29         @fn             CConstIterator (CNodePtr p = nullptr) noexcept
  30         @brief          The default constructor of the const_iterator for the CList class.
  31         @param[in] p    The the node we want to use to construct the iterator.
  32     */
  33     CConstIterator (CNodePtr p = nullptr) noexcept : node (p)
  34     {
  35     }
  36 
  37     /*!
  38         @fn             CConstIterator (const CConstIterator &)
  39         @brief          The copy-constructor for the const_iterator.
  40     */
  41     CConstIterator (const CConstIterator &) = default;
  42 
  43     /*!
  44        @fn             operator= (const CConstIterator &) noexcept
  45        @brief          The default operator= for the iterator.
  46        @return CConstIterator The new value of the const_iterator.
  47     */
  48     CConstIterator &operator= (const CConstIterator &) noexcept = default;
  49 
  50     /*!
  51        @fn operator== (const CConstIterator &other) const noexcept
  52        @brief The  operator == who compare the two node and return true if equal and false otherwise.
  53        @param[in] other The value we want to compare.
  54        @return  bool If the two iterator are equal.
  55     */
  56     bool operator== (const CConstIterator &other) const noexcept
  57     {
  58         return node == other.node;
  59     }
  60 
  61     /*!
  62        @fn operator!= (const CConstIterator &other) const noexcept
  63        @brief The  operator != who compare the two node and return true if not equal and false otherwise.
  64        @param[in] other The value we want to compare.
  65        @return  bool If the two iterator are not equal.
  66     */
  67     bool operator!= (const CConstIterator &other) const noexcept
  68     {
  69         return node != other.node;
  70     }
  71 
  72     /*!
  73        @fn operator++ () noexcept
  74        @brief The  operator ++ who pre-increment the iterator, pass to the next node.
  75        @return  CConstterator The iterator with the new value.
  76     */
  77     CConstIterator &operator++ () noexcept
  78     {
  79         node = node->getNext ();
  80         return *this;
  81     }
  82 
  83     /*!
  84        @fn operator-- () noexcept
  85        @brief The  operator -- who pre-decrement the iterator, pass to the previous node.
  86        @return  CConstterator The iterator with the new value.
  87     */
  88     CConstIterator &operator-- () noexcept
  89     {
  90         node = node->getPrevious ();
  91         return *this;
  92     }
  93 
  94     /*!
  95       @fn operator++ (int) noexcept
  96       @brief The  operator ++ who post-increment the iterator, pass to the next node.
  97       @return  CConstCIterator The iterator with the old value, but iterator are increment.
  98     */
  99     CConstIterator operator++ (int) noexcept
 100     {
 101         CConstIterator temp = *this;
 102         node = node->getNext ();
 103         return temp;
 104     }
 105 
 106     /*!
 107        @fn operator-- (int) noexcept
 108        @brief The  operator -- who post-decrement the iterator, pass to the previous node.
 109        @return  CConstterator The iterator with the old value, but iterator are decrement.
 110     */
 111     CConstIterator operator-- (int) noexcept
 112     {
 113         CConstIterator temp = *this;
 114         node = node->getPrevious ();
 115         return temp;
 116     }
 117 
 118     /*!
 119        @fn operator-> () const noexcept
 120        @brief The dereferencement operator -> who return the a pointer to the info of the node.
 121        @return  CIterBase<T>::pointer The pointer to the value of the node.
 122     */
 123     typename CConstIterBase<T>::pointer operator-> () const noexcept
 124     {
 125         return &node->getInfo ();
 126     }
 127 
 128     /*!
 129        @fn operator* () const snoexcept
 130        @brief The dereferencement operator * who return the a reference to the info of the node.
 131        @return  CIterBase<T>::reference The reference to the value of the node.
 132     */
 133     typename CConstIterBase<T>::reference operator* () const noexcept
 134     {
 135         return node->getInfo ();
 136     }
 137 
 138     /*!
 139        @fn getNode () noexcept
 140        @brief The function return the node of the pointer.
 141        @return  CNodePtr The node of the pointer.
 142     */
 143     CNodePtr getNode () const noexcept
 144     {
 145         return node;
 146     }
 147 };
