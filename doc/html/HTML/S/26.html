 237 /*!
 238     @fn CList<T>::resize (unsigned n, const T &val) noexcept
 239     @brief This function resize at n size, it add a default value (val) if n is greater than the CList size
 240            and delete the old values if n is smaller than the CList size
 241     @param[in] n is @p unsigned parameter which indicates the new size of the CList
 242     @param[in] val is @p T parameter which indicates the default value if n is greater than the CList size
 243  */
 244 template<typename T>
 245 void CList<T>::resize (unsigned n, const T &val /*= T()*/) noexcept
 246 {
 247     if (m_size == n)
 248         return;
 249 
 250     if (m_size > n)
 251     {
 252         CNodePtr ptr = m_head;
 253         for (size_t i = 0; i < n; ++i)
 254             ptr = ptr->getNext ();
 255 
 256         ptr->setNext (m_tail);
 257         m_tail->setPrevious (ptr);
 258 
 259         m_size = n;
 260 
 261         return;
 262     }
 263 
 264     CNodePtr ptr = m_head;
 265     for (size_t i = 0; i < m_size; ++i)
 266         ptr = ptr->getNext ();
 267 
 268     for (size_t i = 0; i < n - m_size; ++i)
 269     {
 270         CNodePtr add = make_shared<CNode> (val, ptr, ptr->getNext ());
 271         ptr->setNext (add);
 272         m_tail->setPrevious (add);
 273         ptr = add;
 274     }
 275 
 276     m_size = n;
 277 }
