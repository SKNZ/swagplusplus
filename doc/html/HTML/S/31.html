  21     // C++11 c'est bon, mangez-en.
  22     template<typename T> using CCollection = std::vector<T>;

  24     template<typename T>
  25     class CValueProvider
  26     {
  27     public:
  28         CCollection<T> operator() (const int /*valueCount*/ = 0)
  29         {
  30             // typeid(T).name() only gives mangled class name, not very clear but still a good indication.
  31             throw std::runtime_error (std::string ("Value provider for type ") + typeid (T).name () + " does not exist.");
  32         }
  33     };

  35     template<typename T>
  36     class CValueProvider<T *>
  37     {
  38     public:
  39         CCollection<T *> operator() (const int valueCount = 0)
  40         {
  41             CCollection<T> array = CValueProvider<T> () (valueCount);
  42             CCollection<T *> ptrArray;
  43             ptrArray.reserve (array.size ());
  44 
  45             for (T x : array)
  46                 ptrArray.push_back (new T (x));
  47 
  48             return ptrArray;
  49         }
  50     };

  55     public:

  59             CCollection<std::shared_ptr<T>> ptrArray;
