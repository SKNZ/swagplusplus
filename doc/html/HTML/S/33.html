 220 template<typename T>
 221 void nsSdD::CList<T>::resize (unsigned n, const T &val /*= T()*/) noexcept
 222 {
 223     if (m_size == n)
 224         return;
 225 
 226     if (0 == n)
 227         clear ();
 228 
 229     if (m_size > n) // Reducing the CList to n
 230     {
 231         iterator itr = begin ();
 232         std::advance (itr, n - 1);
 233 
 234         itr.getNode ()->setNext (m_tail);
 235         m_tail->setPrevious (itr.getNode ());
 236 
 237         m_size = n;
 238     }
 239     else // Expanding the CList to n
 240     {
 241         for (; n != 0; --n)
 242             push_back (val);
 243     }
